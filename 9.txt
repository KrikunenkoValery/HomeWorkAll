#include <stdio.h>
#include <locale.h>

int main() 
{
    // Задание 1
    setlocale(LC_ALL, "RUS");
    for (int i = 0; i <= 5; i++) { 
        for (int j = i; j >= 0; j--) { 
            printf("%d ", j);
        }
        printf("\n");
    }

    // Задание 2
    while (1) 
    {
        long long int num;
        printf("Введите число: ");
        scanf("%lld", &num);

        int digits[20];
        int count = 0;

        while (num > 0) 
        {
            digits[count] = num % 10;
            num /= 10;
            count++;
        }

        for (int i = 0; i < count - 1; i++) 
        {
            if (digits[i] == digits[i + 1]) 
            {
                printf("Совпадающие цифры: %d\n", digits[i]);
            }
        }
        printf("Продолжить? (Да - y, нет - n): ");
        char a;
        while ((a = getchar()) != '\n');
        a = getchar();
        if (a == 'n' || a == 'N') break;
    }


}
