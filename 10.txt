#define M_PI 3.14159
// Задание 3
float area_round(float r)
{
	return M_PI * r * r;
}

void draw_round(float radius, char symbol)
{
	int r = (int)radius;
	for (int i = -r; i <= r; i++)
	{
		for (int j = -r; j <= r; j++)
		{
			if (i * i + j * j <= r * r)
			{
				printf("%c", symbol);
			}
			else {
				printf(" ");
			}
		}
		printf("\n");
	}
}

// Задание 2
double factorial(int n)
{
	if (n == 0) return 1;
	double f = 1.0;
	for (int i = 1; i <= n; i++) {
		f *= i;
	}
	return f;
}
double sin_n(double x, int N)
{
	double sum = 0.0;
	for (int k = 0; k < N; k++) {
		double term = (pow(-1, (k - 1)) * pow(x, 2 * k + 1)) / factorial(2 * k + 1);
		sum += term;
	}
	return sum;
}

double sin_eps(double x, double eps)
{
	double sum = 0.0, term;
	int k = 0;
	do {
		term = (pow(-1, k - 1) * pow(x, 2 * k + 1)) / factorial(2 * k + 1);
		sum += term;
		k++;
	} while (fabs(term) > eps);
	return sum;
}

double fa(double x)
{
	double y1;
	if (x <= 3)
		y1 = pow(x, 2) - 3 * x + 9;
	else
		y1 = 1 / (pow(x, 3) + 3);
	return y1;
}
double fa2(double x, double e)
{
	double y2;
	y2 = x * (pow(e, pow(sin(x), 2)));
	return y2;
}
